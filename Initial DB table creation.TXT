CREATE TABLE users (
	username		 varchar(32),
	email	 varchar(32),
	confirmed	 boolean,
	save_game	 text,
	password_hash varchar(64),
	PRIMARY KEY(username)
);

CREATE TABLE levels (
	id	 int auto_increment,
	level_info	 text NOT NULL,
	name varchar(32) NOT NULL,
	user_name varchar(32) NOT NULL,
	PRIMARY KEY(id)
);

CREATE TABLE score (
	completion_time int,
	collected_coins int,
	enemies_killed	 int,
	level_id	 int,
	user_name	 varchar(32)
);

CREATE TABLE sprite (
	type	 varchar(16) NOT NULL,
	image_name	 varchar(64),
	user_name varchar(32) NOT NULL,
	PRIMARY KEY(image_name)
);

CREATE TABLE votes (
    vote int,
    user_name varchar(32),
    level_id int
    PRIMARY KEY(user_name, level_id)
);
